	\begin{frame}
	\frametitle{Постановка задачи}
	Дана популяция $P$, имеющая размер $N = |P|$, выборка $S$ объемом $N_S = |P|$ с нарушенным условием репрезентативности, $S \subset P$. \\
	
	Необходимо: \\
	
		\begin{itemize}
		\item<1-> Разделить рассматриваемое население $P$ на возможные категории: $S_{(\cdot)}$, $I_{(\cdot \cdot)}$.
		\item<1-> Сформулировать математическую модель.
		\item<1-> Оценить доли здоровых $s_{(\cdot)}$ и ВИЧ-инфицированных $i_{(\cdot \cdot)}$ индивидов.
		\item<1-> Оценить количество людей в различных состояниях $S_{(\cdot)}$, $I_{(\cdot \cdot)}$ для всей популяции $P$.
		\item<1-> Осуществить программную реализацию модели.
		\end{itemize}
	
	\end{frame}
	
	\begin{frame}
	\frametitle{Математическая модель}
	\framesubtitle{Демографический баланс}
	
	\begin{itemize}
		\item<1-> Все население делится на несколько групп
		\item<1-> Все индивиды в группе считаются одинаковыми
		\item<1-> Существуют переходы между группами, входной поток и выходной 
	\end{itemize}
	Динамика i-й группы описывается следующим уравнением: \\
	
	\begin{equation} \label{eqfirst}
	\dot{X_i} = \sum_{i \neq j} (\alpha_{ij}(X) - \alpha_{ji}(X)) - \alpha_{ii}(X) + \omega_i, \notag
	\end{equation}
	
	где \\
	$X_i$ - количество человек в $i$-ой группе, \\
	$\alpha_{ij}(X)$ - скорость перехода из группы $i$ в группу $j$, \\
	$\alpha_{ii}(X)$ - отток из $i$-й группы, $\omega_i$ - приток в $i$-ю группу. \\
	
	Предполагается, что выполнены следующие условия:
	
	A1. $\alpha_{ij}(X) \geq 0$ и $\omega_i(X) \geq 0$ для всех $i$, $j$.
	
	A2. Из $X_i = 0$ следует  $\alpha_{ij}(X)$ и $\alpha_{ii}(X) = 0$.
	\end{frame}

	\begin{frame}
	\frametitle{Математическая модель}
	\framesubtitle{Простейшая модель распространения инфекции}
	\begin{gather}
	\dot{S} = \alpha - (\phi(X) + \mu)S, \notag \\	
	\dot{I_{A}} = \phi(X)S - (\mu + \delta_A)I_{A}, \notag\\
	\dot{I_{C}} = \delta_A I_{A} - (\mu + \delta_C)I_{C},\notag
	\end{gather}
	
	где \\
	$\phi(X) = {{\beta_A I_A + \beta_C I_C}\over{N}}$, \\
	$X \! = \! \left[S\;\;I_{A}\;\;I_{C}\right]$, \\
	$N = S + I_A + I_C$, \\
	$\beta_A$, $\beta_C$ - вероятность заражения за одно взаимодействие для остро и хронически инфицированного соответственно.
			
	\end{frame}

	\begin{frame}
	\frametitle{Математическая модель}
	\framesubtitle{Модель с оценкой поведения}
	
	\begin{gather}
	\dot{S_H} = \alpha_H - (\phi_H(X) + \rho_H + \mu)S_H + \rho_L S_{L,} \notag \\	
	\dot{S_L} = \alpha_L - (\phi_L(X) + \rho_L + \mu)S_L + \rho_H S_{H,}  \notag \\
	\dot{I_{CH}} = \delta_A I_{AH} - (\rho_H + \mu + \delta_C)I_{CH} + \rho_L I_{CL,} \notag \\
	\dot{I_{CL}} = \delta_A I_{AL} - (\rho_L + \mu + \delta_C)I_{CL} + \rho_H I_{CH,} \notag \\
	\dot{I_{AH}} = \phi_H(X)S_H - (\rho_H + \mu + \delta_A)I_{AH} + \rho_L I_{AL,} \notag \\
	\dot{I_{AL}} = \phi_L(X)S_L - (\rho_L + \mu + \delta_A)I_{AL} + \rho_H I_{AH,} \notag
	\end{gather}
	где $X \! = \! \left[I_{CH}\;\;I_{CL}\;\;I_{AH}\;\;I_{AL}\;\;S_H\;\;S_L\right]'$ - вектор переменных состояния, \\
	$\phi_H(X)$ и $\phi_L(X)$ - коэффициенты заболеваемости,
	$\delta_A$ - коэффициент перехода заболевания из острой формы в хроническую, \\
	$\delta_C$ - коэффициент смертности хронически инфицированных из-за СПИДа. \\
	\end{frame}
	
	\begin{frame}
	\frametitle{Выборка и оценка состояний}


	Вероятность, что случайный высокорисковый (низкорисковый соответственно) человек $R$ находится в HRE в случайный момент времени в течение периода отбора:
	
	\begin{gather}
	p_H = P(HRE | R = H), \notag \\
	p_L = P(HRE | R = L). \notag
	\end{gather}
	
	\end{frame}

	\begin{frame}
	\frametitle{Выборка и оценка состояний}
	
	Вероятность случайного индивида $R$, находящегося в HRE в случайный момент времени, относиться к одному из состояний, соответствующих категории высокого риска, т.е. находиться в $X \in Z_H = \{S_H, I_{AH}, I_{CH}\}$, вычисляется с помощью формулы Байеса:
	
	\begin{gather}
	P(X | HRE)= {{P(HRE | R = H)P(X)}\over{P(HRE | R = H)P(R = H) + P(HRE | R = L)P(R = L)}} \notag \\
	= {{p_H {{X} \over {N}}} \over {p_H {{N_H} \over {N}} + p_L {{N_L} \over {N}}}} = {{r_b X} \over {r_b N_H + N_L}},\notag 
	\end{gather}
	
	где $N_H = S_H + I_{AH} + I_{CH}, \; N_L = S_L + I_{AL} + I_{CL}, \; N = N_H + N_L$ и $rb = p_H / p_L$.
	\end{frame}	
	
	\begin{frame}
	\frametitle{Вычисление долей в общей популяции}
	
	Пусть $N_S$ - количество индивидов, которые были отобраны во время посещения HRE в единицу времени:
	
	\begin{gather}
	N_S = {\hat{S}_H} + {\hat{S}_L} + {\hat{I}_{AH}} + {\hat{I}_{AL}} + {\hat{I}_{CH}} + {\hat{I}_{CL}}. \notag
	\end{gather}
	Обозначим число лиц в соответствующих категориях через $\hat{S}_{(.)}$ и $\hat{I}_{(.)}$. Обозначим известные доли соответствующих групп внутри выборки $\hat{s}_{(.)}$ и $\hat{i}_{(.)}$:
	
	\begin{gather}
	{\hat{s}_H} = {{\hat{S}_H} \over N_S}, \; {\hat{s}_L} = {{\hat{S}_L} \over N_S}, \; {\hat{i}_{AH}} = {{\hat{I}_{AH}} \over N_S}, \; {\hat{i}_{AL}} = {{\hat{I}_{AL}} \over N_S}, \; {\hat{i}_{CH}} = {{\hat{I}_{CH}} \over N_S}, \; {\hat{i}_{CL}} = {{\hat{I}_{CL}} \over N_S}. \notag
	\end{gather}
	
	\end{frame}		
	
	
	\begin{frame}
	\frametitle{Вычисление долей в общей популяции}
	\framesubtitle{Случай для 6 групп}	
	\begin{gather} \label{restr1}
	\begin{cases}
	s_H r_b	= \hat{s}_H r_b (s_H + i_{AH} +i_{CH}) + \hat{s}_H(s_L + i_{AL} +i_{CL}), \notag \\
	s_L	= \hat{s}_L r_b (s_H + i_{AH} +i_{CH}) + \hat{s}_L(s_L + i_{AL} +i_{CL}), \notag \\
	i_{AH} r_b	= \hat{i}_{AH} r_b (s_H + i_{AH} +i_{CH}) + \hat{i}_{AH}(s_L + i_{AL} +i_{CL}),  \\
	i_{AL} = \hat{i}_{AL} r_b (s_H + i_{AH} +i_{CH}) + \hat{i}_{AL}(s_L + i_{AL} +i_{CL}),  \notag \\
	i_{CH} r_b	= \hat{i}_{CH} r_b (s_H + i_{AH} +i_{CH}) + \hat{i}_{CH}(s_L + i_{AL} +i_{CL}),  \notag \\
	i_{CL} = \hat{i}_{CL} r_b (s_H + i_{AH} +i_{CH}) + \hat{i}_{CL}(s_L + i_{AL} +i_{CL}),  \notag \\
	{s}_H + {s}_L + {i}_{AH} + {i}_{AL} + {i}_{CH} + {i}_{CL} = 1. \notag
	\end{cases}
	\end{gather}
	\end{frame}
	
	\begin{frame}
	\frametitle{Вычисление долей в общей популяции}
	\framesubtitle{Общий случай}
	
	
	\begin{equation}
	\begin{bmatrix}
	\begin{pmatrix}
	r_b I_{\left[ H \right]} & 0\\
	0& I_{\left[ L \right]}
	\end{pmatrix}
	+ (1 - r_b)
	\begin{pmatrix}
	diag(\hat{x}_H) & 0\\
	0& diag(\hat{x}_L)
	\end{pmatrix}
	\begin{pmatrix}
	\textbf{1} & \textbf{0}\\
	\textbf{1} & \textbf{0}
	\end{pmatrix}
	\end{bmatrix}
	\begin{pmatrix}
	x_H\\
	x_L
	\end{pmatrix}
	=
	\begin{pmatrix}
	\hat{x}_H\\
	\hat{x}_L
	\end{pmatrix}, \notag
	\end{equation}
	
	$x_H = (s_H \; i_{CH})^\mathrm{T}$, $x_L = (i_{CL})^\mathrm{T}$, \\
	$\hat{x}_H = (\hat{s}_H \; \hat{i}_{CH})^\mathrm{T}$, $\hat{x}_L = (\hat{i}_{CL})^\mathrm{T}$, \\
	$s_L = 1 - s_H - i_{CH} - i_{CL}$.\\
	\end{frame}

	\begin{frame}
	\frametitle{Заключение}
	
	\begin{itemize}
		\item<1-> Анализ существующих эпидемиологических моделей распространения инфекции.
		\item<1-> Математическая модель, представляющая популяцию как совокупность шести различных групп, учитывающая два уровня рискованного поведения и различные стадии инфицирования рассматриваемого населения: острая и хроническая.
		\item<1-> Оценка соотношения категорий людей в выборке для нерепрезентативных данных, основываясь на их поведении.
		\item<1-> Вычисление долей в общей популяции для конкретного примера четырех состояний и для общего случая.
		\item<1-> Программная реализация и численный эксперимент для конкретной выборки.
	\end{itemize}
